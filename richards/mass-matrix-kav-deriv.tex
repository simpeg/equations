\begin{equation}
\label{eq:mass-matrix-kav-deriv}
\deriv{\mathbf{k}_{Av}(\boldsymbol{\psi})}{\boldsymbol{\psi}}
=
\text{diag}\left(
    (\mathbf{A}_v \mathbf{k}^{-1}(\boldsymbol{\psi}))^{-2} \right)
\mathbf{A}_v
\text{diag} \left(
    \mathbf{k}^{-2}(\boldsymbol{\psi}) \right)
\deriv{\mathbf{k}(\boldsymbol{\psi})}{\boldsymbol{\psi}}
\end{equation}
